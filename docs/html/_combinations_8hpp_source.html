37  std::vector<unsigned long> &pos, unsigned long depth,
38  unsigned long margin, std::vector<std::vector<int> >& combos) {
39  // Have we selected the number of required elements?
40  if (depth >= req_len) {
41  std::vector<int> v(pos.size());
42  for (unsigned long ii = 0; ii < pos.size(); ++ii) {
43  v[ii] = elems[pos[ii]]; //.push_back(elems[pos[ii]]);
44  }
45  combos.push_back(v);
46  return;
47  }

50  for (unsigned long ii = margin; ii < elems.size(); ++ii) {
51  pos[depth] = ii;
52  Combonations::CombinationsRRecursive(elems, req_len, pos, depth + 1, ii, combos);
