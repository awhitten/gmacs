306  const T coefficients(const T &x, const T &sel_coeffs)
307  {
308  int x1 = x.indexmin();
309  int x2 = x.indexmax();
310  int y2 = sel_coeffs.indexmax();
311  T y(x1,x2);
312  for ( int i = x1; i < y2; i++ )

314  y(i) = exp(sel_coeffs(i)) / (1.0 + exp(sel_coeffs(i)));

317  return y;
