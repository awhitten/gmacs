122 void nloglike::tail_compression()
123 {
124 
125  double pmin = 0.001;
126 
127 
128  jmin.allocate(r1,r2);
129  jmax.allocate(r1,r2);
130 
131  for(int i = r1; i <= r2; i++ )
132  {
133 
134  dvector o = m_O(i);
135 
136  jmin(i) = c1+1; // index for min column
137  jmax(i) = c2; // index for max column
138  dvector cumsum = o/sum(o);
139  for(int j = c1+1; j <= c2; j++ )
140  {
141  cumsum(j) += cumsum(j-1);
142  cumsum(j) <= pmin ? jmin(i)++ : NULL;
143  j != c2 ? 1.0 - cumsum(j) < pmin ? jmax(i)-- : NULL : NULL;
144  }
145 
146  }
147 
148  // now allocate ragged arrays
149  m_Or.allocate(r1,r2,jmin,jmax);
150  m_Pr.allocate(r1,r2,jmin,jmax);
151  m_Or.initialize();
152  m_Pr.initialize();
153 
154  // fill ragged arrays
155  for(int i = r1; i <= r2; i++ )
156  {
157  dvector cso = m_O(i)/sum(m_O(i));
158  dvar_vector csp = m_P(i)/sum(m_P(i));
159  m_Or(i)(jmin(i),jmax(i)) = cso(jmin(i),jmax(i));
160  m_Pr(i)(jmin(i),jmax(i)) = csp(jmin(i),jmax(i));
161 
162  // add cumulative sum to tails.
163  m_Or(i)(jmin(i)) = sum(cso(c1,jmin(i)));
164  m_Or(i)(jmax(i)) = sum(cso(jmax(i),c2));
165  m_Pr(i)(jmin(i)) = sum(csp(c1,jmin(i)));
166  m_Pr(i)(jmax(i)) = sum(csp(jmax(i),c2));
